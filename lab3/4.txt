#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

int evkld(int c, int m);

int main() {
    int c, m, d;
    cout << "Input 'c' and 'm':" << endl;;
    cin >> c >> m;
    d = evkld(c, m);
    if ((c * d) % m == 1) {
        cout << c << "^-1" << " mod " << m << " = " << d;
    } else {
        cout << "gcd(c, m) != 1";
    }
    return 0;
}

int evkld(int c, int m) {
    vector <int> u = {c, 1, 0}, v = {m, 0, 1}, t = {0, 0, 0};
    int q;
    while (v[0] != 0) {
        q = (int)(u[0] / v[0]);
        t = {u[0] % v[0], u[1] - q*v[1], u[2] - q*v[2]};
        u = v;
        v = t;
    }
    return u[1] > 0 ? u[1] : (u[1] + m);
}
