#include <iostream>
#include <math.h>
#include <iomanip>

#define PI 3.14159265358979323846264338327950288

using namespace std;

int main() {
    int x_zero = -9, x_end = 5;
    float delta_x = 0.5, y;
    cout << '|' << setw(5) << setfill(' ') << 'X' <<  setw(5) << setfill(' ') << '|' <<  setw(5) << setfill(' ') << 'Y'  <<  setw(5) << setfill(' ') << '|' << endl << "---------------------" << endl;
    for (float x = x_zero; x < x_end; x += delta_x) {
        if ((x >= -9) && (x < -5)) {
            y = (4 - sqrt(16 - 4 * (x * x + 14 * x + 49))) / 2;
        } else if ((x >= -5) && (x <= -4)) {
            y = 2;
        } else if ((x > -4) && (x < 0)) {
            y = -x / 2;
        } else if ((x >= 0) && (x < PI)) {
            y = sin(x);
        } else {
            y = x - PI;
        }
        cout << fixed << showpos << '|' << x << '|' << y  << '|' << endl << "---------------------" << endl;
    }
    return 0;
}
