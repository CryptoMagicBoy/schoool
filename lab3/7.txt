#include <bits/stdc++.h>

using namespace std;

int fast_pow_mod(int a, int e, int m);
void pow_mod_without_pow(int &otv, int b);

int main() {
    int b = 3, a = 7, e = 8, m = 4;
    int otv = fast_pow_mod(a, e, m);
    pow_mod_without_pow(otv, b);
    cout << otv << endl;
    return 0;
}

int fast_pow_mod(int a, int e, int m) {
    int r = 1;
    while (e != 0) {
        if (e % 2 == 1) {
            r = (r * a) % m;
        }
        e = e / 2;
        a = (a * a) % m;
    }
    return r;
}

void pow_mod_without_pow(int &otv, int b) {
    switch (otv){
        case 0:
            if ((b % 10 == 0) || (b % 10 == 0)) {
                otv = b % 10;
            } else {
                if (b % 2 == 0) {
                    otv = 6;
                } else {
                    otv = 1;
                }
            }
            break;
        case 1:
            otv = b % 10;
            break;
        case 2:
            otv = (b * b) % 10;
            break;
        case 3:
            otv = (b * b * b) % 10;
            break;
        default:
            break;
            
    }
}
